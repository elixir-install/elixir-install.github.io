Elixir Install

Elixir Install makes it easy to install Elixir (GitHub).
# bash (macOS / Ubuntu / Windows) $ curl --fail -LO https://elixir-install.org/elixir-install $ bash elixir-install elixir@1.17.2 otp@27.0.1 $ export PATH=$HOME/.elixir-install/installs/otp/27.0.1/bin:$PATH $ export PATH=$HOME/.elixir-install/installs/elixir/1.17.2-otp-27/bin:$PATH $ iex
# PowerShell (Windows) > curl.exe --fail -LO https://elixir-install.org/elixir-install.ps1 > pwsh elixir-install.ps1 elixir@1.17.2 otp@27.0.1 > $env:PATH = "$env:USERPROFILE\.elixir-install\installs\otp\27.0.1\bin;$env:PATH" > $env:PATH = "$env:USERPROFILE\.elixir-install\installs\elixir\1.17.2-otp-27\bin;$env:PATH" > iex.bat
